# 12: Subplots
# Generate a single figure containing four subplots as follows:
# sin(x), cos(x), tan(x), 1.0/(1.0  + exp(-x))

x=np.linspace(0,2*np.pi)
plt.figure(figsize=(8,8))

y1=np.sin(x)
plt.subplot(2,2,1)
plt.plot(x,y1)
plt.title("sin(x)")


y2=np.cos(x)
plt.subplot(2,2,2)
plt.plot(x,y2)
plt.title("cos(x)")

y3=np.tan(x)
plt.subplot(2,2,3)
plt.plot(x,y3)
plt.title("tan(x)")

y4=1.0/(1.0  + np.exp(-x))
plt.subplot(2,2,4)
plt.plot(x,y4)
plt.title("Sigmoid")

plt.show()

# 14: Pie Chart
# You are given data on the number of cars sold by a dealership in a month for four different brands: Toyota, Ford, Honda, and Tesla. The numbers are as follows:
#
# Toyota: 45 cars
# Ford: 30 cars
# Honda: 15 cars
# Tesla: 10 cars
# Create a pie chart using Matplotlib to visualize the share of monthly car sales for each brand. Make sure to include:

# Labels for each brand
# Percentages on the pie chart
# A title
# Different colors for each section

brand=np.array(['Toyota','Ford','Honda','Tesla'])
num_sold=np.array([45,30,15,10])
colors=np.array(['orange','blue','red','green'])

plt.pie(num_sold,labels=brand,autopct='%1.1f%%',colors=colors)
plt.title("Brands vs Number of Cars Sold")
plt.show()
